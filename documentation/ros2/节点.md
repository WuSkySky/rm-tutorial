##### 最小实现
```py
import rclpy
from rclpy.node import Node

class NODE_CLASS(Node):
    def __init__(self, name):
        super().__init__(name)

def main(args=None):
    rclpy.init(args=args)
    node = NODE_CLASS(NODE_NAME)
    try:
        rclpy.spin(node)
    except KeyboardInterrupt:
        pass
    finally:
        node.destroy_node()
```
